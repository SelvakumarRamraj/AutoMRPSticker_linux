Public Class DbUpgrade

    Public Shared Sub Run()
        Dim currentVersion As Integer =
            SqlHelper.ExecuteScalar(Conn, CommandType.Text,
                "SELECT ISNULL(MAX(VersionNo),0) FROM sys.DBVersion")

        For v As Integer = currentVersion + 1 To 100
            Dim file = Path.Combine(Application.StartupPath,
                        "DB\Upgrade_" & v & ".sql")

            If File.Exists(file) Then
                Dim sql = File.ReadAllText(file)
                SqlHelper.ExecuteNonQuery(Conn, CommandType.Text, sql)
            End If
        Next
    End Sub
End Class

VB.NET – AUTO UPDATE CHECK
--------------------------------------------------
Public Class AutoUpdater

    Public Shared Sub Check()
        Dim localVer = Application.ProductVersion
        Dim remoteVer As String

        Using wc As New Net.WebClient()
            remoteVer = wc.DownloadString("https://yourserver.com/erp/version.txt").Trim()
        End Using

        If remoteVer <> localVer Then
            DownloadPatch(remoteVer)
        End If
    End Sub
End Class

**************************************************
PATCH DOWNLOAD & APPLY
--------------------------------------------------
Private Shared Sub DownloadPatch(ver As String)
    Dim patchUrl = $"https://yourserver.com/erp/patch_{ver}.zip"
    Dim zipPath = Application.StartupPath & "\patch.zip"

    My.Computer.Network.DownloadFile(patchUrl, zipPath)
    ZipFile.ExtractToDirectory(zipPath, Application.StartupPath, True)
    Application.Restart()
End Sub

Public Shared Sub Validate()

    Dim ok = SqlHelper.ExecuteScalar(Conn, CommandType.Text,
        "SELECT 1 FROM sys.License

Public Shared Sub Backup()
    Dim file = "ERP_" & Today.ToString("yyyyMMdd") & ".bak"

    SqlHelper.ExecuteNonQuery(Conn, CommandType.Text,
        "BACKUP DATABASE ERPDB TO DISK=@F",
        New SqlParameter("@F", "C:\Backup\" & file))

    UploadToCloud("C:\Backup\" & file)
End Sub

Private Shared Sub UploadToCloud(path As String)
    Dim req = CType(Net.WebRequest.Create("ftp://server/backup/" &
               Path.GetFileName(path)), Net.FtpWebRequest)

    req.Method = Net.WebRequestMethods.Ftp.UploadFile
    req.Credentials = New Net.NetworkCredential("user", "pwd")

    Dim bytes = File.ReadAllBytes(path)
    Using s = req.GetRequestStream()
        s.Write(bytes, 0, bytes.Length)
    End Using
End Sub

Public Class LicenseController
    Inherits Controller

    <HttpPost>
    Function Activate(key As String, machineId As String) As JsonResult
        Dim ok = SqlHelper.ExecuteScalar(Conn, CommandType.Text,
            "SELECT 1 FROM sys.LicenseMaster
             WHERE LicenseKey=@K AND IsUsed=0",
            New SqlParameter("@K", key))

        If ok Is Nothing Then
            Return Json(New With {.success = False, .msg = "Invalid Key"})
        End If

        SqlHelper.ExecuteNonQuery(Conn, CommandType.Text,
            "UPDATE sys.LicenseMaster
             SET MachineID=@M, IsUsed=1, ActivatedOn=GETDATE()
             WHERE LicenseKey=@K",
            New SqlParameter("@M", machineId),
            New SqlParameter("@K", key))

        Return Json(New With {.success = True})
    End Function
End Class
