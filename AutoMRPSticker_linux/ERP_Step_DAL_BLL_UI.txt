Imports System.Data.SqlClient

Public Class SqlHelper

    Public Shared Function ExecuteDataTable(
        connStr As String,
        cmdType As CommandType,
        cmdText As String,
        ParamArray params() As SqlParameter) As DataTable

        Using con As New SqlConnection(connStr)
            Using cmd As New SqlCommand(cmdText, con)
                cmd.CommandType = cmdType
                If params IsNot Nothing Then cmd.Parameters.AddRange(params)
                Using da As New SqlDataAdapter(cmd)
                    Dim dt As New DataTable
                    da.Fill(dt)
                    Return dt
                End Using
            End Using
        End Using
    End Function

    Public Shared Function ExecuteScalar(
        connStr As String,
        cmdType As CommandType,
        cmdText As String,
        ParamArray params() As SqlParameter) As Object

        Using con As New SqlConnection(connStr)
            Using cmd As New SqlCommand(cmdText, con)
                cmd.CommandType = cmdType
                If params IsNot Nothing Then cmd.Parameters.AddRange(params)
                con.Open()
                Return cmd.ExecuteScalar()
            End Using
        End Using
    End Function

    Public Shared Sub ExecuteNonQuery(
        connStr As String,
        cmdType As CommandType,
        cmdText As String,
        ParamArray params() As SqlParameter)

        Using con As New SqlConnection(connStr)
            Using cmd As New SqlCommand(cmdText, con)
                cmd.CommandType = cmdType
                If params IsNot Nothing Then cmd.Parameters.AddRange(params)
                con.Open()
                cmd.ExecuteNonQuery()
            End Using
        End Using
    End Sub

End Class

Public Class LoginBLL

    Public Shared Function ValidateUser(
        userName As String,
        password As String) As Boolean

        Dim sql = "SELECT UserID FROM sec.Users 
                   WHERE UserName=@U AND PasswordHash=@P AND IsActive=1"

        Dim dt = SqlHelper.ExecuteDataTable(
            My.Settings.ERPConn,
            CommandType.Text,
            sql,
            New SqlParameter("@U", userName),
            New SqlParameter("@P", password))

        If dt.Rows.Count = 1 Then
            SessionManager.UserID = dt.Rows(0)("UserID")
            SessionManager.UserName = userName
            Return True
        End If
        Return False
    End Function

End Class

        Dim sql = "

        FROM sec.Menu m
        JOIN sec.RoleMenu rm ON rm.MenuID = m.MenuID

**************************************BLL – Save Sales
Public Class SalesBLL

    Public Shared Sub Save(docNo As String, docDate As Date,
                           custId As Integer, dtLines As DataTable)

        If Not FinancialYearBLL.IsDateAllowed(docDate) Then
            Throw New Exception("Financial Year Locked")
        End If

        Dim docId = SqlHelper.ExecuteScalar(
            My.Settings.ERPConn,
            CommandType.Text,
            "INSERT INTO trx.SalesH(DocNo,DocDate,CustomerID,NetAmount)
             VALUES(@N,@D,@C,0); SELECT SCOPE_IDENTITY()",
            New SqlParameter("@N", docNo),
            New SqlParameter("@D", docDate),
            New SqlParameter("@C", custId))

        For Each r As DataRow In dtLines.Rows
            ' Save line
            SqlHelper.ExecuteNonQuery(
                My.Settings.ERPConn,
                CommandType.Text,
                "INSERT INTO trx.SalesL(DocID,ItemID,Qty,Rate)
                 VALUES(@D,@I,@Q,@R)",
                New SqlParameter("@D", docId),
                New SqlParameter("@I", r("ItemID")),
                New SqlParameter("@Q", r("Qty")),
                New SqlParameter("@R", r("Rate")))

            ' Stock OUT
            StockBLL.Post(r("ItemID"), docDate, 0, r("Qty"), r("Rate"), docNo)
        Next

    End Sub

End Class

***********************************Moving Avg Calculation
Public Class StockBLL

    Public Shared Sub Post(itemId As Integer, docDate As Date,
                           inQty As Decimal, outQty As Decimal,
                           rate As Decimal, refDoc As String)

        ' Current stock
        Dim dt = SqlHelper.ExecuteDataTable(
            My.Settings.ERPConn,
            CommandType.Text,
            "SELECT SUM(InQty-OutQty) Qty, SUM(TransValue) Val
             FROM trx.StockLedger WHERE ItemID=@I",
            New SqlParameter("@I", itemId))

        Dim curQty = If(IsDBNull(dt.Rows(0)("Qty")), 0, dt.Rows(0)("Qty"))
        Dim curVal = If(IsDBNull(dt.Rows(0)("Val")), 0, dt.Rows(0)("Val"))

        Dim newVal = (curVal + inQty * rate)
        Dim newQty = (curQty + inQty)

        Dim avgRate = If(newQty = 0, rate, newVal / newQty)

        SqlHelper.ExecuteNonQuery(
            My.Settings.ERPConn,
            CommandType.Text,
            "INSERT INTO trx.StockLedger
             (ItemID,DocDate,InQty,OutQty,Rate,TransValue,RefDoc)
             VALUES(@I,@D,@IN,@OUT,@R,@V,@REF)",
            New SqlParameter("@I", itemId),
            New SqlParameter("@D", docDate),
            New SqlParameter("@IN", inQty),
            New SqlParameter("@OUT", outQty),
            New SqlParameter("@R", avgRate),
            New SqlParameter("@V", (inQty - outQty) * avgRate),
            New SqlParameter("@REF", refDoc))
    End Sub

End Class

*****************************************Profit & Loss View
CREATE VIEW rpt.vw_PnL AS
SELECT a.AccName,
       SUM(l.Cr - l.Dr) Amount
FROM acc.Ledger l
JOIN acc.Account a ON a.AccID = l.AccID
WHERE a.AccType IN ('Income','Expense')
GROUP BY a.AccName

*******************************************SALES GST POSTING (BLL)
Public Class GSTBLL

    Public Shared Sub PostSalesGST(docDate As Date, taxable As Decimal, gstId As Integer, ref As String)

        Dim dt = SqlHelper.ExecuteDataTable(
            My.Settings.ERPConn,
            CommandType.Text,
            "SELECT * FROM acc.GSTSlab WHERE GSTID=@ID",
            New SqlParameter("@ID", gstId))

        Dim cgst = taxable * dt.Rows(0)("CGST") / 100
        Dim sgst = taxable * dt.Rows(0)("SGST") / 100
        Dim igst = taxable * dt.Rows(0)("IGST") / 100

        AccountsBLL.Post(201, docDate, 0, cgst, ref) ' CGST Output
        AccountsBLL.Post(202, docDate, 0, sgst, ref)
        AccountsBLL.Post(203, docDate, 0, igst, ref)

    End Sub
End Class

********************************************TDS POSTING (PURCHASE)
Public Class TDSBLL

    Public Shared Sub PostTDS(docDate As Date, amount As Decimal, tdsId As Integer, ref As String)

        Dim rate = SqlHelper.ExecuteScalar(
            My.Settings.ERPConn,
            CommandType.Text,
            "SELECT TDSRate FROM acc.TDSSlab WHERE TDSID=@I",
            New SqlParameter("@I", tdsId))

        Dim tdsAmt = amount * rate / 100

        AccountsBLL.Post(301, docDate, 0, tdsAmt, ref) ' TDS Payable
        AccountsBLL.Post(401, docDate, tdsAmt, 0, ref) ' Expense Reduction

    End Sub
End Class

******************************************YEAR-END CLOSING AUTOMATION (CRITICAL)
🔹 Year Close Procedure
--------------------------------
CREATE PROCEDURE acc.usp_YearEndClose
@OldFY INT,
@NewFY INT
AS
BEGIN
    -- Transfer balances
    INSERT INTO acc.Ledger (AccID, DocDate, Dr, Cr, RefDoc)
    SELECT AccID, GETDATE(),
           CASE WHEN SUM(Dr-Cr) > 0 THEN SUM(Dr-Cr) ELSE 0 END,
           CASE WHEN SUM(Dr-Cr) < 0 THEN ABS(SUM(Dr-Cr)) ELSE 0 END,
           'FY OPENING'
    FROM acc.Ledger
    GROUP BY AccID

    UPDATE mst.FinancialYear SET IsLocked=1 WHERE FYID=@OldFY
END
