CREATE TABLE Roles (
  RoleId INT IDENTITY PRIMARY KEY,
  RoleName NVARCHAR(50),
  IsSuperAdmin nvarchar(1) 'Y'
);


CREATE TABLE Menus (
    MenuCode NVARCHAR(50) PRIMARY KEY,
    MenuName NVARCHAR(100),
    Url NVARCHAR(200)
);


CREATE TABLE RoleMenus (
    RoleId INT NOT NULL,
    MenuCode NVARCHAR(50) NOT NULL,
    CONSTRAINT PK_RoleMenus PRIMARY KEY (RoleId, MenuCode),
    CONSTRAINT FK_RoleMenus_Roles FOREIGN KEY (RoleId) REFERENCES Roles(RoleId),

    UserId INT IDENTITY PRIMARY KEY,

CREATE TABLE Warehouse (
    WhId INT IDENTITY PRIMARY KEY,
    Whscode nvarchar(30) Not Null,
    WhName NVARCHAR(50) NOT NULL
);

CREATE TABLE Stock (
    ItemId INT NOT NULL,
    WhId INT NOT NULL,
    Qty DECIMAL(18,2) DEFAULT 0,
    AvgPrice DECIMAL(18,2) DEFAULT 0,
    CONSTRAINT PK_Stock PRIMARY KEY(ItemId, WhId),
    CONSTRAINT FK_Stock_Item FOREIGN KEY (ItemId) REFERENCES Item(ItemId),
    CONSTRAINT FK_Stock_Warehouse FOREIGN KEY (WhId) REFERENCES Warehouse(WhId)
);



CREATE TABLE StockLedger (
    LedgerId INT IDENTITY PRIMARY KEY,
    ItemId INT NOT NULL,
    WhId INT NOT NULL,
    ObjType nvarchar(20),
    DocType NVARCHAR(20),
    DocNo NVARCHAR(30),
    Docentry int, 
    DocDate DATE,
    InQty DECIMAL(18,2) DEFAULT 0,
    OutQty DECIMAL(18,2) DEFAULT 0,
    Price decimal(10,2),
    Avgprice Decimal(10,2),
    StockValue decimal(18,2),    
    CONSTRAINT FK_StockLedger_Item FOREIGN KEY (ItemId) REFERENCES Item(ItemId),
    CONSTRAINT FK_StockLedger_Warehouse FOREIGN KEY (WhId) REFERENCES Warehouse(WhId)
    
);
CREATE INDEX IDX_StockLedger_ItemId ON StockLedger(ItemId);
CREATE INDEX IDX_StockLedger_WhId ON StockLedger(WhId);
