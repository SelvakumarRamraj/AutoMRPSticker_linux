Imports System.Data.SqlClient


Public Class SalesDAL


Public Shared Function SaveSales(userId As Integer,
items As DataTable) As Boolean
Using con As SqlConnection = SqlHelper.GetConnection()
Dim trans As SqlTransaction = con.BeginTransaction()


Try
' Insert Master
Dim cmdMaster As String =
"INSERT INTO SalesMaster(SalesDate,UserId)
VALUES(GETDATE(),@UserId);
SELECT SCOPE_IDENTITY();"


Dim salesId As Integer
Using cmd As New SqlCommand(cmdMaster, con, trans)
cmd.Parameters.AddWithValue("@UserId", userId)
salesId = Convert.ToInt32(cmd.ExecuteScalar())
End Using


' Insert Details
For Each r As DataRow In items.Rows
Dim cmdDetail As String =
"INSERT INTO SalesDetail(SalesId,ItemId,Qty,Rate)
VALUES(@SalesId,@ItemId,@Qty,@Rate)"


SqlHelper.ExecuteNonQuery(cmdDetail, trans,
New SqlParameter("@SalesId", salesId),
New SqlParameter("@ItemId", r("ItemId")),
New SqlParameter("@Qty", r("Qty")),
New SqlParameter("@Rate", r("Rate")))
Next


trans.Commit()
Return True


Catch ex As Exception
trans.Rollback()
Throw
End Try
End Using
End Function


End Class

Public Shared Sub SavePurchase(h As DataRow, dt As DataTable)

    Using con = SqlHelper.GetConnection()
        Dim tr = con.BeginTransaction()

        Try
            Dim pid As Integer =
              SqlHelper.ExecuteScalar(
                "INSERT INTO PurchaseHeader(PDate,PartyId)
                 VALUES(@D,@P); SELECT SCOPE_IDENTITY()",
                tr,
                SqlHelper.P("@D", h("Date")),
                SqlHelper.P("@P", h("PartyId")))

            For Each r As DataRow In dt.Rows

                SqlHelper.ExecuteNonQuery(
                  "INSERT INTO PurchaseDetail VALUES(@H,@I,@Q,@R)",
                  tr,
                  SqlHelper.P("@H", pid),
                  SqlHelper.P("@I", r("ItemId")),
                  SqlHelper.P("@Q", r("Qty")),
                  SqlHelper.P("@R", r("Rate")))

                ' STOCK UPDATE
                SqlHelper.ExecuteNonQuery(
                  "MERGE Stock t
                   USING (SELECT @I ItemId,@Q Qty) s
                   ON t.ItemId=s.ItemId
                   WHEN MATCHED THEN UPDATE SET Qty=Qty+@Q
                   WHEN NOT MATCHED THEN INSERT VALUES(@I,@Q)",
                  tr,
                  SqlHelper.P("@I", r("ItemId")),
                  SqlHelper.P("@Q", r("Qty")))
            Next

            tr.Commit()

        Catch
            tr.Rollback()
            Throw
        End Try
    End Using
End Sub

********************************************SqlHelper
Public Class SqlHelper

    Private Shared ReadOnly conStr As String =
        ConfigurationManager.ConnectionStrings("ERP").ConnectionString

    Public Shared Function GetConnection() As SqlConnection
        Dim con As New SqlConnection(conStr)
        con.Open()
        Return con
    End Function

    Public Shared Function P(name As String, value As Object) As SqlParameter
        Return New SqlParameter(name, If(value, DBNull.Value))
    End Function

    Public Shared Function ExecuteScalar(sql As String, tr As SqlTransaction,
                                         ParamArray p() As SqlParameter) As Object
        Using cmd As New SqlCommand(sql, tr.Connection, tr)
            cmd.Parameters.AddRange(p)
            Return cmd.ExecuteScalar()
        End Using
    End Function

    Public Shared Sub ExecuteNonQuery(sql As String, tr As SqlTransaction,
                                      ParamArray p() As SqlParameter)
        Using cmd As New SqlCommand(sql, tr.Connection, tr)
            cmd.Parameters.AddRange(p)
            cmd.ExecuteNonQuery()
        End Using
    End Sub

    Public Shared Function ExecuteDataTable(sql As String) As DataTable
        Using con = GetConnection()
            Using da As New SqlDataAdapter(sql, con)
                Dim dt As New DataTable
                da.Fill(dt)
                Return dt
            End Using
        End Using
    End Function

End Class

Public Class DocNoDAL

    Public Shared Function GetNextNo(docType As String,
                                     tr As SqlTransaction) As String

        Dim sql As String =
        "UPDATE DocumentSeries WITH (UPDLOCK, HOLDLOCK)
         SET LastNo = LastNo + 1
         OUTPUT INSERTED.Prefix + CAST(INSERTED.LastNo AS NVARCHAR)
         WHERE DocType = @T"

        Return SqlHelper.ExecuteScalar(sql, tr,
               SqlHelper.P("@T", docType)).ToString()

    End Function

End Class

Public Class PostingPeriodDAL

    Public Shared Sub Validate(docDate As DateTime, tr As SqlTransaction)

        Dim sql As String =
        "SELECT COUNT(*) FROM PostingPeriod
         WHERE @D BETWEEN FromDate AND ToDate AND IsOpen=1"

        Dim cnt As Integer =
            Convert.ToInt32(SqlHelper.ExecuteScalar(sql, tr,
                SqlHelper.P("@D", docDate)))

        If cnt = 0 Then
            Throw New ApplicationException("Posting period closed")
        End If

    End Sub

*******************************************save sales

Public Shared Function SaveSales(h As DataRow,
                                 d As DataTable) As Integer

    Using con = SqlHelper.GetConnection()
        Dim tr = con.BeginTransaction()

        Try
            ' PERIOD CHECK
            PostingPeriodDAL.Validate(h("SalesDate"), tr)

            ' DOC NUMBER
            Dim invNo As String =
                DocNoDAL.GetNextNo("SALES", tr)

            Dim sid As Integer =
                SqlHelper.ExecuteScalar(
                "INSERT INTO Sales_Header
                 (InvoiceNo,InvoiceDate,PartyId,Total)
                 VALUES(@N,@D,@P,@T);
                 SELECT SCOPE_IDENTITY()",
                tr,
                SqlHelper.P("@N", invNo),
                SqlHelper.P("@D", h("SalesDate")),
                SqlHelper.P("@P", h("PartyId")),
                SqlHelper.P("@T", h("Total")))

            For Each r As DataRow In d.Rows

                ' STOCK OUT
                SqlHelper.ExecuteNonQuery(
                "UPDATE Stock SET Qty = Qty - @Q
                 WHERE ItemId=@I AND Qty >= @Q",
                tr,
                SqlHelper.P("@Q", r("Qty")),
                SqlHelper.P("@I", r("ItemId")))

                ' PREVENT NEGATIVE STOCK
                If SqlHelper.ExecuteScalar(
                    "SELECT @@ROWCOUNT", tr) = 0 Then
                    Throw New ApplicationException("Insufficient stock")
                End If

                SqlHelper.ExecuteNonQuery(
                "INSERT INTO Sales_Detail VALUES(@H,@I,@Q,@R)",
                tr,
                SqlHelper.P("@H", sid),
                SqlHelper.P("@I", r("ItemId")),
                SqlHelper.P("@Q", r("Qty")),
                SqlHelper.P("@R", r("Rate")))
            Next

            tr.Commit()
            Return sid

        Catch
            tr.Rollback()
            Throw
        End Try
    End Using
End Function

*******************************************sales cancel logic
Public Shared Sub CancelSales(salesId As Integer)

    Using con = SqlHelper.GetConnection()
        Dim tr = con.BeginTransaction()

        Try
            Dim dt = SqlHelper.ExecuteDataTable(
                "SELECT ItemId,Qty FROM Sales_Detail WHERE SalesId=" & salesId)

            ' RESTORE STOCK
            For Each r As DataRow In dt.Rows
                SqlHelper.ExecuteNonQuery(
                "UPDATE Stock SET Qty = Qty + @Q WHERE ItemId=@I",
                tr,
                SqlHelper.P("@Q", r("Qty")),
                SqlHelper.P("@I", r("ItemId")))
            Next

            ' MARK CANCELLED
            SqlHelper.ExecuteNonQuery(
                "UPDATE Sales_Header SET IsCancelled=1 WHERE SalesId=@I",
                tr,
                SqlHelper.P("@I", salesId))

            tr.Commit()

        Catch
            tr.Rollback()
            Throw
        End Try
    End Using
End Sub

*******************************************Item Transfer DAL
Public Class ItemTransferDAL

    Public Shared Sub Save(h As DataRow, d As DataTable)

        Using con = SqlHelper.GetConnection()
            Dim tr = con.BeginTransaction()

            Try
                ' Posting period
                PostingPeriodDAL.Validate(h("TransferDate"), tr)

                ' Document No
                Dim docNo = DocNoDAL.GetNextNo("TRANSFER", tr)

                Dim tid =
                    SqlHelper.ExecuteScalar(
                    "INSERT INTO ItemTransfer_Header
                     (TransferNo,TransferDate,FromWh,ToWh)
                     VALUES(@N,@D,@F,@T);
                     SELECT SCOPE_IDENTITY()",
                    tr,
                    SqlHelper.P("@N", docNo),
                    SqlHelper.P("@D", h("TransferDate")),
                    SqlHelper.P("@F", h("FromWh")),
                    SqlHelper.P("@T", h("ToWh")))

                For Each r As DataRow In d.Rows

                    ' STOCK OUT (FROM WH)
                    SqlHelper.ExecuteNonQuery(
                    "UPDATE Stock SET Qty = Qty - @Q
                     WHERE ItemId=@I AND WhId=@W AND Qty>=@Q",
                    tr,
                    SqlHelper.P("@Q", r("Qty")),
                    SqlHelper.P("@I", r("ItemId")),
                    SqlHelper.P("@W", h("FromWh")))

                    If SqlHelper.ExecuteScalar("SELECT @@ROWCOUNT", tr) = 0 Then
                        Throw New ApplicationException("Insufficient stock")
                    End If

                    ' STOCK IN (TO WH)
                    SqlHelper.ExecuteNonQuery(
                    "MERGE Stock t
                     USING (SELECT @I ItemId,@W WhId,@Q Qty) s
                     ON t.ItemId=s.ItemId AND t.WhId=s.WhId
                     WHEN MATCHED THEN UPDATE SET Qty=Qty+@Q
                     WHEN NOT MATCHED THEN INSERT VALUES(@I,@W,@Q)",
                    tr,
                    SqlHelper.P("@I", r("ItemId")),
                    SqlHelper.P("@W", h("ToWh")),
                    SqlHelper.P("@Q", r("Qty")))

                    SqlHelper.ExecuteNonQuery(
                        "INSERT INTO ItemTransfer_Detail VALUES(@H,@I,@Q)",
                        tr,
                        SqlHelper.P("@H", tid),
                        SqlHelper.P("@I", r("ItemId")),
                        SqlHelper.P("@Q", r("Qty")))
                Next

                tr.Commit()

            Catch
                tr.Rollback()
                Throw
            End Try
        End Using
    End Sub

End Class

********************************Insert Ledger
Public Shared Sub PostLedger(tr As SqlTransaction,
                             itemId As Integer,
                             whId As Integer,
                             docType As String,
                             docNo As String,
                             docDate As Date,
                             inQty As Decimal,
                             outQty As Decimal)

    SqlHelper.ExecuteNonQuery(
    "INSERT INTO StockLedger
     (ItemId,WhId,DocType,DocNo,DocDate,InQty,OutQty)
     VALUES(@I,@W,@T,@N,@D,@IN,@OUT)",
    tr,
    SqlHelper.P("@I", itemId),
    SqlHelper.P("@W", whId),
    SqlHelper.P("@T", docType),
    SqlHelper.P("@N", docNo),
    SqlHelper.P("@D", docDate),
    SqlHelper.P("@IN", inQty),
    SqlHelper.P("@OUT", outQty))
End Sub

***************************************************Stock Report
SELECT DocDate, DocType, DocNo,
       InQty, OutQty,
       SUM(ISNULL(InQty,0)-ISNULL(OutQty,0))
       OVER (ORDER BY DocDate, LedgerId) AS Balance
FROM StockLedger
WHERE ItemId=@ItemId AND WhId=@WhId
ORDER BY DocDate, LedgerId

Public Shared Sub UpdateStockOnPurchase(itemId As Integer,
                                        whId As Integer,
                                        purchaseQty As Decimal,
                                        purchaseRate As Decimal,
                                        tr As SqlTransaction)

    ' Get current stock
    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT Qty, AvgPrice FROM Stock WHERE ItemId=@I AND WhId=@W",
        tr,
        SqlHelper.P("@I", itemId),
        SqlHelper.P("@W", whId)
    )

    If dt.Rows.Count = 0 Then
        ' New stock
        SqlHelper.ExecuteNonQuery(
            "INSERT INTO Stock(ItemId, WhId, Qty, AvgPrice) VALUES(@I,@W,@Q,@P)",
            tr,
            SqlHelper.P("@I", itemId),
            SqlHelper.P("@W", whId),
            SqlHelper.P("@Q", purchaseQty),
            SqlHelper.P("@P", purchaseRate)
        )
    Else
        ' Existing stock: Weighted Average
        Dim curQty As Decimal = dt.Rows(0)("Qty")
        Dim curAvg As Decimal = dt.Rows(0)("AvgPrice")

        Dim newQty As Decimal = curQty + purchaseQty
        Dim newAvg As Decimal = (curQty * curAvg + purchaseQty * purchaseRate) / newQty

        SqlHelper.ExecuteNonQuery(
            "UPDATE Stock SET Qty=@Q, AvgPrice=@P WHERE ItemId=@I AND WhId=@W",
            tr,
            SqlHelper.P("@Q", newQty),
            SqlHelper.P("@P", newAvg),
            SqlHelper.P("@I", itemId),
            SqlHelper.P("@W", whId)
        )
    End If
End Sub

Public Shared Function UpdateStockOnSale(itemId As Integer,
                                         whId As Integer,
                                         saleQty As Decimal,
                                         tr As SqlTransaction) As Decimal

    ' Get stock and average price
    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT Qty, AvgPrice FROM Stock WHERE ItemId=@I AND WhId=@W",
        tr,
        SqlHelper.P("@I", itemId),
        SqlHelper.P("@W", whId)
    )

    If dt.Rows.Count = 0 OrElse dt.Rows(0)("Qty") < saleQty Then
        Throw New ApplicationException("Insufficient stock")
    End If

    Dim curQty As Decimal = dt.Rows(0)("Qty")
    Dim avgPrice As Decimal = dt.Rows(0)("AvgPrice")

    Dim newQty As Decimal = curQty - saleQty

    SqlHelper.ExecuteNonQuery(
        "UPDATE Stock SET Qty=@Q WHERE ItemId=@I AND WhId=@W",
        tr,
        SqlHelper.P("@Q", newQty),
        SqlHelper.P("@I", itemId),
        SqlHelper.P("@W", whId)
    )

    ' Return avg price for gross profit calculation
    Return avgPrice
End Function

Public Shared Sub SaveSalesInvoice(h As DataRow, d As DataTable)

    Using con = SqlHelper.GetConnection()
        Dim tr = con.BeginTransaction()
        Try
            ' Posting period check
            PostingPeriodDAL.Validate(h("SalesDate"), tr)

            ' Invoice number
            Dim invNo As String = DocNoDAL.GetNextNo("SALES", tr)

            Dim salesId As Integer = SqlHelper.ExecuteScalar(
                "INSERT INTO Sales_Header(InvoiceNo, InvoiceDate, PartyId, Total) VALUES(@N,@D,@P,@T); SELECT SCOPE_IDENTITY()",
                tr,
                SqlHelper.P("@N", invNo),
                SqlHelper.P("@D", h("SalesDate")),
                SqlHelper.P("@P", h("PartyId")),
                SqlHelper.P("@T", h("Total"))
            )

            For Each r As DataRow In d.Rows
                ' Update stock & get avg price
                Dim avgPrice As Decimal = UpdateStockOnSale(r("ItemId"), r("WhId"), r("Qty"), tr)

                ' Gross profit per item
                Dim grossProfit As Decimal = r("Rate") * r("Qty") - avgPrice * r("Qty")

                ' Insert sales detail
                SqlHelper.ExecuteNonQuery(
                    "INSERT INTO Sales_Detail(SalesId, ItemId, Qty, Rate, AvgPrice, GrossProfit) VALUES(@H,@I,@Q,@R,@A,@G)",
                    tr,
                    SqlHelper.P("@H", salesId),
                    SqlHelper.P("@I", r("ItemId")),
                    SqlHelper.P("@Q", r("Qty")),
                    SqlHelper.P("@R", r("Rate")),
                    SqlHelper.P("@A", avgPrice),
                    SqlHelper.P("@G", grossProfit)
                )

                ' Post ledger
                LedgerDAL.PostLedger(tr, r("ItemId"), r("WhId"), "SALES", invNo, h("SalesDate"), 0, r("Qty") * avgPrice)
            Next

            tr.Commit()
        Catch
            tr.Rollback()
            Throw
        End Try
    End Using
End Sub

            SqlHelper.P("@W", whId)
        )
    End If
End Sub

Public Enum StockTransactionType
    Purchase          ' Adding stock from purchase
    PurchaseCancel    ' Cancelling purchase invoice
    Sale              ' Reducing stock due to sale
    SaleCancel        ' Cancelling sale invoice
End Enum

Public Shared Sub UpdateStock(itemId As Integer,
                              whId As Integer,
                              qty As Decimal,
                              rate As Decimal,
                              tr As SqlTransaction,
                              transType As StockTransactionType)

    ' Get current stock
    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT Qty, AvgPrice FROM Stock WHERE ItemId=@I AND WhId=@W",
        tr,
        SqlHelper.P("@I", itemId),
        SqlHelper.P("@W", whId)
    )

    Dim curQty As Decimal = 0
    Dim curAvg As Decimal = 0

    If dt.Rows.Count > 0 Then
        curQty = dt.Rows(0)("Qty")
        curAvg = dt.Rows(0)("AvgPrice")
    End If

    Dim newQty As Decimal = curQty
    Dim newAvg As Decimal = curAvg

    Select Case transType
        Case StockTransactionType.Purchase
            ' Add purchase: update weighted average
            newQty = curQty + qty
            newAvg = (curQty * curAvg + qty * rate) / newQty

        Case StockTransactionType.PurchaseCancel
            ' Cancel purchase: remove purchased qty and value
            newQty = curQty - qty
            If newQty > 0 Then
                newAvg = ((curQty * curAvg) - (qty * rate)) / newQty
            Else
                newAvg = 0
            End If

        Case StockTransactionType.Sale
            ' Sale: reduce quantity, avg price remains
            newQty = curQty - qty

        Case StockTransactionType.SaleCancel
            ' Cancel sale: add back sold qty, avg price remains
            newQty = curQty + qty
    End Select

    ' Insert new stock if doesn't exist
    If dt.Rows.Count = 0 Then
        SqlHelper.ExecuteNonQuery(
            "INSERT INTO Stock(ItemId, WhId, Qty, AvgPrice) VALUES(@I,@W,@Q,@P)",
            tr,
            SqlHelper.P("@I", itemId),
            SqlHelper.P("@W", whId),
            SqlHelper.P("@Q", newQty),
            SqlHelper.P("@P", newAvg)
        )
    Else
        ' Update existing stock
        SqlHelper.ExecuteNonQuery(
            "UPDATE Stock SET Qty=@Q, AvgPrice=@P WHERE ItemId=@I AND WhId=@W",
            tr,
            SqlHelper.P("@Q", newQty),
            SqlHelper.P("@P", newAvg),
            SqlHelper.P("@I", itemId),
            SqlHelper.P("@W", whId)
        )
    End If
End Sub

CREATE TABLE SalesDetail (
    DetailId INT IDENTITY PRIMARY KEY,
    SalesId INT NOT NULL,
    ItemId INT NOT NULL,

    Qty DECIMAL(18,2) NOT NULL,
    Rate DECIMAL(18,2) NOT NULL,

    DiscPer DECIMAL(5,2) DEFAULT 0,
    Discount AS (
        ROUND(Qty * Rate * DiscPer / 100, 2)
    ) PERSISTED,

    TaxableValue AS (
        ROUND((Qty * Rate) - Discount, 2)
    ) PERSISTED,

    CGSTRate DECIMAL(5,2) DEFAULT 0,
    SGSTRate DECIMAL(5,2) DEFAULT 0,
    IGSTRate DECIMAL(5,2) DEFAULT 0,

    CGST AS (ROUND(TaxableValue * CGSTRate / 100, 2)) PERSISTED,
    SGST AS (ROUND(TaxableValue * SGSTRate / 100, 2)) PERSISTED,

    LineTotal AS (
        ROUND(TaxableValue + CGST + SGST + IGST, 2)
    ) PERSISTED,

    WhId INT NOT NULL
);

Public Shared Sub PostVoucher(vType As String,
                              vNo As String,
                              vDate As Date,
                              narration As String,
                              entries As List(Of Tuple(Of Integer, Decimal, Decimal)),
                              tr As SqlTransaction)

    SqlHelper.ExecuteNonQuery(
        "INSERT INTO Voucher_Header(VoucherType,VoucherNo,VoucherDate,Narration)
         VALUES(@T,@N,@D,@R)",
        tr,
        SqlHelper.P("@T", vType),
        SqlHelper.P("@N", vNo),
        SqlHelper.P("@D", vDate),
        SqlHelper.P("@R", narration)
    )

    Dim vid As Integer = SqlHelper.ExecuteScalar("SELECT SCOPE_IDENTITY()", tr)

    For Each e In entries
        SqlHelper.ExecuteNonQuery(
            "INSERT INTO Voucher_Detail(VoucherId,AccountId,Debit,Credit)
             VALUES(@V,@A,@D,@C)",
            tr,
            SqlHelper.P("@V", vid),
            SqlHelper.P("@A", e.Item1),
            SqlHelper.P("@D", e.Item2),
            SqlHelper.P("@C", e.Item3)
        )
    Next
End Sub

Public Shared Sub PostPurchaseGRPO(grpoId As Integer, tr As SqlTransaction)

    Dim hdr = SqlHelper.ExecuteDataRow(
        "SELECT GRPONo, GRPODate, POId FROM GRPO_Header WHERE GRPOId=@I",
        tr, SqlHelper.P("@I", grpoId))

    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT D.ItemId, D.Qty, P.Rate, PO.PartyId
         FROM GRPO_Detail D
         JOIN PO_Detail P ON D.ItemId=P.ItemId
         JOIN PO_Header PO ON PO.POId=P.POId
         WHERE D.GRPOId=@I",
        tr, SqlHelper.P("@I", grpoId))

    Dim totalValue As Decimal = 0

    For Each r As DataRow In dt.Rows
        Dim value = r("Qty") * r("Rate")
        totalValue += value

        ' Update Stock
        UpdateStock(r("ItemId"), 1, r("Qty"), r("Rate"), tr, StockTransactionType.Purchase)

        ' Stock Ledger
        SqlHelper.ExecuteNonQuery(
            "INSERT INTO StockLedger(ItemId,WhId,DocType,DocNo,DocDate,InQty)
             VALUES(@I,1,'GRPO',@N,@D,@Q)",
            tr,
            SqlHelper.P("@I", r("ItemId")),
            SqlHelper.P("@N", hdr("GRPONo")),
            SqlHelper.P("@D", hdr("GRPODate")),
            SqlHelper.P("@Q", r("Qty")))
    Next

    ' ===== Accounting =====
    PostVoucher(
        "Purchase",
        hdr("GRPONo"),
        hdr("GRPODate"),
        "Purchase GRPO",
        New List(Of Tuple(Of Integer, Decimal, Decimal)) From {
            Tuple.Create(Accounts.Inventory, totalValue, 0D),
            Tuple.Create(Accounts.SupplierControl, 0D, totalValue)
        },
        tr)

End Sub

Public Shared Sub PostSales(salesId As Integer, tr As SqlTransaction)

    Dim hdr = SqlHelper.ExecuteDataRow(
        "SELECT SalesId, SalesDate FROM SalesMaster WHERE SalesId=@I",
        tr, SqlHelper.P("@I", salesId))

    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT D.ItemId,D.Qty,D.Rate,D.WhId,S.AvgPrice
         FROM SalesDetail D
         JOIN Stock S ON S.ItemId=D.ItemId AND S.WhId=D.WhId
         WHERE D.SalesId=@I",
        tr, SqlHelper.P("@I", salesId))

    Dim salesValue As Decimal = 0
    Dim costValue As Decimal = 0

    For Each r As DataRow In dt.Rows
        Dim salesAmt = r("Qty") * r("Rate")
        Dim costAmt = r("Qty") * r("AvgPrice")

        salesValue += salesAmt
        costValue += costAmt

        ' Update Stock
        UpdateStock(r("ItemId"), r("WhId"), r("Qty"), 0, tr, StockTransactionType.Sale)

        ' Stock Ledger
        SqlHelper.ExecuteNonQuery(
            "INSERT INTO StockLedger(ItemId,WhId,DocType,DocNo,DocDate,OutQty)
             VALUES(@I,@W,'SALE',@N,@D,@Q)",
            tr,
            SqlHelper.P("@I", r("ItemId")),
            SqlHelper.P("@W", r("WhId")),
            SqlHelper.P("@N", salesId),
            SqlHelper.P("@D", hdr("SalesDate")),
            SqlHelper.P("@Q", r("Qty")))
    Next

    ' ===== Accounting =====
    PostVoucher("Sales", salesId, hdr("SalesDate"), "Sales Invoice",
        New List(Of Tuple(Of Integer, Decimal, Decimal)) From {
            Tuple.Create(Accounts.CustomerControl, salesValue, 0D),
            Tuple.Create(Accounts.Sales, 0D, salesValue),
            Tuple.Create(Accounts.COGS, costValue, 0D),
            Tuple.Create(Accounts.Inventory, 0D, costValue)
        }, tr)

End Sub

**************************************cancel voucher
Public Shared Sub CancelVoucher(voucherNo As String, tr As SqlTransaction)

    Dim vid = SqlHelper.ExecuteScalar(
        "SELECT VoucherId FROM Voucher_Header WHERE VoucherNo=@N",
        tr, SqlHelper.P("@N", voucherNo))

    ' Reverse voucher
    SqlHelper.ExecuteNonQuery(
        "INSERT INTO Voucher_Header(VoucherType,VoucherNo,VoucherDate,Narration)
         SELECT VoucherType,VoucherNo+'-CAN',GETDATE(),'Cancel'
         FROM Voucher_Header WHERE VoucherId=@V",
        tr, SqlHelper.P("@V", vid))

    Dim newVid = SqlHelper.ExecuteScalar("SELECT SCOPE_IDENTITY()", tr)

    SqlHelper.ExecuteNonQuery(
        "INSERT INTO Voucher_Detail(VoucherId,AccountId,Debit,Credit)
         SELECT @NV,AccountId,Credit,Debit
         FROM Voucher_Detail WHERE VoucherId=@V",
        tr,
        SqlHelper.P("@NV", newVid),
        SqlHelper.P("@V", vid))

    SqlHelper.ExecuteNonQuery(
        "UPDATE Voucher_Header SET IsCancelled=1 WHERE VoucherId=@V",
        tr, SqlHelper.P("@V", vid))
End Sub

Public Shared Sub PostSalesAgainstSO(salesId As Integer, tr As SqlTransaction)

    Dim dt = SqlHelper.ExecuteDataTable(
        "SELECT ItemId, Qty, Rate, WhId, SOId
         FROM SalesDetail WHERE SalesId=@I",
        tr, SqlHelper.P("@I", salesId))

    For Each r As DataRow In dt.Rows

        ' Reduce open qty in SO
        SqlHelper.ExecuteNonQuery(
            "UPDATE SalesOrder_Detail
             SET OpenQty = OpenQty - @Q
             WHERE SOId=@S AND ItemId=@I",
            tr,
            SqlHelper.P("@Q", r("Qty")),
            SqlHelper.P("@S", r("SOId")),
            SqlHelper.P("@I", r("ItemId")))

        ' Update stock
        UpdateStock(
            r("ItemId"),
            r("WhId"),
            r("Qty"),
            0,
            tr,
            StockTransactionType.Sale)

    Next

    ' Update SO status
    SqlHelper.ExecuteNonQuery(
        "UPDATE SalesOrder_Header
         SET Status =
           CASE
             WHEN NOT EXISTS (SELECT 1 FROM SalesOrder_Detail WHERE SOId=@S AND OpenQty>0)
             THEN 'CLOSED'
             ELSE 'PARTIAL'
           END
         WHERE SOId=@S",
        tr, SqlHelper.P("@S", dt.Rows(0)("SOId")))

End Sub

Public Shared Sub PostPurchaseVoucher(
    purchaseId As Integer,
    purchaseDate As Date,
    supplierAcc As Integer,
    subTotal As Decimal,
    cgst As Decimal,
    sgst As Decimal,
    igst As Decimal,
    roundOff As Decimal,
    netAmount As Decimal,
    tr As SqlTransaction)

    Dim vId As Integer = Convert.ToInt32(
        SqlHelper.ExecuteScalar(
            "INSERT INTO VoucherHeader
             (VoucherDate, RefId, VoucherType)
             VALUES(@D,@R,'PURCHASE');
             SELECT SCOPE_IDENTITY()",
            tr,
            SqlHelper.P("@D", purchaseDate),
            SqlHelper.P("@R", purchaseId)
        )
    )

    ' Purchase Expense
    InsertVLine(vId, Accounts.Purchase, subTotal, 0, tr)

    ' GST Input
    If cgst > 0 Then InsertVLine(vId, Accounts.InputCGST, cgst, 0, tr)
    If sgst > 0 Then InsertVLine(vId, Accounts.InputSGST, sgst, 0, tr)
    If igst > 0 Then InsertVLine(vId, Accounts.InputIGST, igst, 0, tr)

    ' Round Off
    If roundOff <> 0 Then
        InsertVLine(
            vId,
            Accounts.RoundOff,
            If(roundOff < 0, Math.Abs(roundOff), 0),
            If(roundOff > 0, roundOff, 0),
            tr
        )
    End If

    ' Supplier
    InsertVLine(vId, supplierAcc, 0, netAmount, tr)
End Sub

Public Shared Sub PostGSTChallan(
    challanDate As Date,
    cgstPay As Decimal,
    sgstPay As Decimal,
    igstPay As Decimal,
    bankAcc As Integer,
    tr As SqlTransaction)

    Dim vId As Integer = Convert.ToInt32(
        SqlHelper.ExecuteScalar(
            "INSERT INTO VoucherHeader
             (VoucherDate, VoucherType)
             VALUES(@D,'GSTCHALLAN');
             SELECT SCOPE_IDENTITY()",
            tr,
            SqlHelper.P("@D", challanDate)
        )
    )

    ' Output GST Dr
    If cgstPay > 0 Then InsertVLine(vId, Accounts.OutputCGST, cgstPay, 0, tr)
    If sgstPay > 0 Then InsertVLine(vId, Accounts.OutputSGST, sgstPay, 0, tr)
    If igstPay > 0 Then InsertVLine(vId, Accounts.OutputIGST, igstPay, 0, tr)

    ' Bank Cr
    InsertVLine(vId, bankAcc, 0, cgstPay + sgstPay + igstPay, tr)
End Sub

DECLARE @AccountId INT = 101;   -- Party / Ledger account

SELECT
    H.VoucherDate,
    H.VoucherType,
    H.RefId,
    V.Debit,
    V.Credit,
    SUM(ISNULL(V.Debit,0) - ISNULL(V.Credit,0))
        OVER (ORDER BY H.VoucherDate, V.VoucherDetailId) AS RunningBalance
FROM VoucherDetail V
JOIN VoucherHeader H ON H.VoucherId = V.VoucherId
WHERE V.AccountId = @AccountId
  AND ISNULL(H.IsCancelled,0)=0
ORDER BY H.VoucherDate, V.VoucherDetailId;

*****************************************ageing report
SELECT
    P.PartyName,
    SUM(CASE WHEN Days <= 30 THEN Balance ELSE 0 END) AS [0-30],
    SUM(CASE WHEN Days BETWEEN 31 AND 60 THEN Balance ELSE 0 END) AS [31-60],
    SUM(CASE WHEN Days BETWEEN 61 AND 90 THEN Balance ELSE 0 END) AS [61-90],
    SUM(CASE WHEN Days > 90 THEN Balance ELSE 0 END) AS [90+]
FROM (
    SELECT
        H.RefId,
        P.PartyName,
        DATEDIFF(DAY, H.VoucherDate, GETDATE()) AS Days,
        SUM(V.Debit - V.Credit) AS Balance
    FROM VoucherHeader H
    JOIN VoucherDetail V ON H.VoucherId=V.VoucherId
    JOIN Party P ON P.AccountId=V.AccountId
    WHERE H.VoucherType IN ('SALES','PURCHASE')
      AND ISNULL(H.IsCancelled,0)=0
    GROUP BY H.RefId, H.VoucherDate, P.PartyName
) X
GROUP BY PartyName;

***********************************Closing financial year
Public Sub CloseFinancialYear(yearId As Integer)

    Dim tr As SqlTransaction = con.BeginTransaction()

    Try
        Dim profit As Decimal = GetNetProfit(yearId, tr)

        Dim vId As Integer = InsertVoucher(
            "YEARCLOSE",
            GetYearEndDate(yearId),
            tr)

        ' P&L Account
        If profit >= 0 Then
            InsertVLine(vId, Accounts.PnL, profit, 0, tr)
            InsertVLine(vId, Accounts.Capital, 0, profit, tr)
        Else
            InsertVLine(vId, Accounts.Capital, Math.Abs(profit), 0, tr)
            InsertVLine(vId, Accounts.PnL, 0, Math.Abs(profit), tr)
        End If

        ' Mark year closed
        UpdateYearClosed(yearId, tr)

        ' Carry forward balances
        CarryForwardBalances(yearId, tr)

        tr.Commit()

    Catch ex As Exception
        tr.Rollback()
        Throw
    End Try

End Sub

;WITH cte AS (
    SELECT *,
           SUM(InQty - OutQty) OVER (
               PARTITION BY ItemId, WhId
               ORDER BY DocDate, LedgerId
           ) AS RunQty,

           SUM(TransValue) OVER (
               PARTITION BY ItemId, WhId
               ORDER BY DocDate, LedgerId
           ) AS RunValue
    FROM StockLedger
)
UPDATE cte
SET BalanceQty   = RunQty,
    BalanceValue = RunValue,
    AvgPrice     = CASE WHEN RunQty = 0 THEN 0 ELSE RunValue / RunQty END;

Private Sub LoadMenus()
    Dim dt As DataTable = SqlHelper.ExecuteDataset(
        conStr,
        CommandType.Text,
        "SELECT MenuCode, MenuName, ParentMenuCode 
         FROM Menus WHERE Active='Y' ORDER BY SortOrder"
    ).Tables(0)

    TreeView1.Nodes.Clear()

    For Each row As DataRow In dt.Select("ParentMenuCode IS NULL")
        Dim parent As New TreeNode(row("MenuName").ToString())
        parent.Tag = row("MenuCode").ToString()
        TreeView1.Nodes.Add(parent)
        LoadChildMenus(parent, dt)
    Next
End Sub

Private Sub LoadChildMenus(parent As TreeNode, dt As DataTable)
    For Each row As DataRow In dt.Select("ParentMenuCode='" & parent.Tag & "'")
        Dim child As New TreeNode(row("MenuName").ToString())
        child.Tag = row("MenuCode").ToString()
        parent.Nodes.Add(child)
        LoadChildMenus(child, dt)
    Next
End Sub

********************************Load Role Permissions (Check TreeView)
Private Sub LoadRoleMenus(roleId As Integer)
    Dim dt As DataTable = SqlHelper.ExecuteDataset(
        conStr,
        CommandType.Text,
        "SELECT MenuCode FROM RoleMenus WHERE RoleId=@RoleId",
        New SqlParameter("@RoleId", roleId)
    ).Tables(0)

    CheckNodes(TreeView1.Nodes, dt)
End Sub

Private Sub CheckNodes(nodes As TreeNodeCollection, dt As DataTable)
    For Each node As TreeNode In nodes
        node.Checked = dt.Select("MenuCode='" & node.Tag & "'").Length > 0
        CheckNodes(node.Nodes, dt)
    Next
End Sub

        New SqlParameter("@RoleId", roleId))

    SaveCheckedNodes(TreeView1.Nodes)

Private Sub SaveCheckedNodes(nodes As TreeNodeCollection)
    For Each node As TreeNode In nodes
        If node.Checked Then
            SqlHelper.ExecuteNonQuery(conStr, CommandType.Text,
                "INSERT INTO RoleMenus VALUES (@RoleId,@MenuCode)",
                New SqlParameter("@RoleId", roleId),
                New SqlParameter("@MenuCode", node.Tag))
        End If
        SaveCheckedNodes(node.Nodes)
    Next
End Sub

----------------------------------AuthController (VB.NET)
Imports System.IdentityModel.Tokens.Jwt
Imports Microsoft.IdentityModel.Tokens
Imports System.Security.Claims
Imports System.Text

Public Class AuthController
    Inherits ApiController

    <HttpPost>
    <Route("api/auth/token")>
    Public Function Token(login As LoginModel) As IHttpActionResult

        ' Validate user
        Dim dt = SqlHelper.ExecuteDataset(conStr, CommandType.Text,
            "SELECT u.UserId,u.UserName,u.RoleId,r.IsSuperAdmin
             FROM Users u
             JOIN Roles r ON u.RoleId=r.RoleId
             WHERE u.UserName=@U AND u.PasswordHash=@P AND u.Active=1",
            New SqlParameter("@U", login.UserName),
            New SqlParameter("@P", Hash(login.Password))
        ).Tables(0)

        If dt.Rows.Count = 0 Then
            Return Unauthorized()
        End If

        Dim row = dt.Rows(0)

        Dim claims = New List(Of Claim) From {
            New Claim("UserId", row("UserId").ToString()),
            New Claim("RoleId", row("RoleId").ToString()),
            New Claim("IsSuperAdmin", row("IsSuperAdmin").ToString())
        }

        Dim key = New SymmetricSecurityKey(Encoding.UTF8.GetBytes("ERP_SUPER_SECRET_KEY"))
        Dim creds = New SigningCredentials(key, SecurityAlgorithms.HmacSha256)

        Dim token = New JwtSecurityToken(
            issuer:="ERP",
            audience:="ERPUsers",
            claims:=claims,
            expires:=DateTime.Now.AddHours(8),
            signingCredentials:=creds
        )

        Return Ok(New With {
            .token = New JwtSecurityTokenHandler().WriteToken(token)
        })
    End Function
End Class

API AUTHORIZATION FILTER (MENU-WISE)
🔹 Custom Attribute
----------------------------
Imports System.IdentityModel.Tokens.Jwt
Imports System.Net
Imports System.Net.Http
Imports System.Web.Http.Controllers
Imports System.Web.Http.Filters

Public Class ApiMenuAuthorizeAttribute
    Inherits AuthorizationFilterAttribute

    Public Property MenuCode As String

    Public Overrides Sub OnAuthorization(ctx As HttpActionContext)

        Dim auth = ctx.Request.Headers.Authorization
        If auth Is Nothing OrElse auth.Scheme <> "Bearer" Then
            ctx.Response = ctx.Request.CreateResponse(HttpStatusCode.Unauthorized)
            Exit Sub
        End If

        Dim handler As New JwtSecurityTokenHandler()
        Dim token = handler.ReadJwtToken(auth.Parameter)

        Dim isSuper = token.Claims.First(Function(c) c.Type = "IsSuperAdmin").Value = "True"
        If isSuper Then Exit Sub

        Dim roleId = token.Claims.First(Function(c) c.Type = "RoleId").Value

        Dim dt = SqlHelper.ExecuteDataset(conStr, CommandType.Text,
            "SELECT 1 FROM RoleMenus WHERE RoleId=@R AND MenuCode=@M",
            New SqlParameter("@R", roleId),
            New SqlParameter("@M", MenuCode)
        ).Tables(0)

        If dt.Rows.Count = 0 Then
            ctx.Response = ctx.Request.CreateResponse(HttpStatusCode.Forbidden)
        End If
    End Sub
End Class

------------------------REFRESH TOKEN API
<HttpPost>
<Route("api/auth/refresh")>
Public Function RefreshToken(model As RefreshRequest) As IHttpActionResult

    Dim dt = SqlHelper.ExecuteDataset(conStr, CommandType.Text,
        "SELECT UserId FROM RefreshTokens
         WHERE RefreshToken=@T AND IsRevoked=0 AND ExpiryDate>GETDATE()",
        New SqlParameter("@T", model.RefreshToken)
    ).Tables(0)

    If dt.Rows.Count = 0 Then
        Return Unauthorized()
    End If

    Dim userId = CInt(dt.Rows(0)("UserId"))

    ' Regenerate Access Token
    Dim userRow = GetUser(userId)
    Dim newAccessToken = GenerateJwt(userRow)

    Return Ok(New With {.access_token = newAccessToken})
End Function

------------------------------LOGOUT / TOKEN REVOKE
<HttpPost>
<Route("api/auth/logout")>
Public Function Logout(model As RefreshRequest) As IHttpActionResult
    SqlHelper.ExecuteNonQuery(conStr, CommandType.Text,
        "UPDATE RefreshTokens SET IsRevoked=1 WHERE RefreshToken=@T",
        New SqlParameter("@T", model.RefreshToken)
    )
    Return Ok()
End Function
